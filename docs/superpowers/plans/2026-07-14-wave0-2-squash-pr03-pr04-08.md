**Goal:** 完成 v0.5.5 Batch 的 Wave 0(squash refactor 分支)+ Wave 1(PR-03 schema_migrations 元表 + SchedulingConfig 字段扩展)+ Wave 2(PR-04~08 五个 unwired 领域包),让 `backend/scheduling/*` 具备完整的编排能力但不 wire 到主链路。

```go
		t.Fatalf("first call: %v", err)
```

```go
package database

import (
	"time"

	"scheduling-system/backend/models"
)

// EnsureMigrationApplied 幂等地记录一个 migration 已应用。
// 已存在同名 version 时不重复插入,不报错。用于 v0.5.5 版本链:
//   v0.5.5-prep    — PR-03,DTO / 元表 / SchedulingMode 字段就绪,行为不变
//   v0.5.5-dbgate  — PR-09,time_assignments 建表 + schedule_entries 拆分
//   v0.5.5-release — PR-10,主链路切至 Orchestrator
//
// 契约:接收 DB 接口(通常是应用启动路径中的 *GormAdapter),
// 通过 FirstOrCreate 保证幂等。
func EnsureMigrationApplied(db DB, name string) error {
	rec := models.SchemaMigration{
		Version:   name,
		AppliedAt: time.Now(),
	}
	// FirstOrCreate:存在则读回,不存在则用 rec 建。
	return db.Where("version = ?", name).FirstOrCreate(&rec).Error()
}
```

Create `scripts/check-deps.sh`:

```bash
#!/usr/bin/env bash
# INV-A4 依赖边界:backend/scheduling/* 不允许依赖 services / database / wails。
# 见 docs/superpowers/specs/2026-07-14-scheduling-dual-mode-batch-execution.md §7。
set -e

FORBIDDEN=(
  "scheduling-system/backend/services"
  "scheduling-system/backend/database"
  "scheduling-system/backend/wails"
)

FAIL=0
for pkg in $(go list ./backend/scheduling/... 2>/dev/null); do
  deps=$(go list -deps -f '{{.ImportPath}}' "$pkg")
  for forbidden in "${FORBIDDEN[@]}"; do
    if echo "$deps" | grep -q "^${forbidden}"; then
      echo "❌ $pkg imports $forbidden"
      FAIL=1
    fi
  done
done

if [ $FAIL -eq 0 ]; then
  echo "✅ backend/scheduling/* 依赖边界正常"
else
  exit 1
fi
```

```go
// 从原 backend/services/resource_matcher.go 提取,不依赖任何 services / database。
//
// 见:
//   - docs/adr/0006-unified-resource-matching-framework.md (URMF R3)

package matcher

import (
	"encoding/json"
	"strings"

	"scheduling-system/backend/models"

// ResourceMatchCode 是匹配结果的分类枚举。
type ResourceMatchCode int

const (
	MatchOK                 ResourceMatchCode = 0
	CodeRoomTypeMismatch    ResourceMatchCode = 1
	CodeEquipmentMissing    ResourceMatchCode = 2
	CodeSpecialtyExclusion  ResourceMatchCode = 3
)

// MatchResult 是匹配返回的统一结构。字段与老 services.MatchResult 完全一致。
type MatchResult struct {
	OK               bool              `json:"ok"`
	Code             ResourceMatchCode `json:"code"`
	Reason           string            `json:"reason,omitempty"`
	RequiredType     string            `json:"requiredType,omitempty"`
	ActualType       string            `json:"actualType,omitempty"`
	MissingEquipment []string          `json:"missingEquipment,omitempty"`
}

// ResourceRequirement 描述任务对教室的资源需求(纯结构化数据)。
type ResourceRequirement struct {
	RoomType       string   `json:"roomType"`
	RoomTypeSource string   `json:"roomTypeSource"`
	Equipment      []string `json:"equipment"`
	IsShared       bool     `json:"isShared"`
}

// Match 检查教室是否满足任务资源需求(硬约束)。
// 从 services/resource_matcher.go 平移,行为等价。
func Match(task models.TeachingTask, course models.Course, room models.Classroom) MatchResult {
	requiredType := InferRoomType(task, course)
	actualType := room.Type

	// (以下逻辑从 services.Match 逐行复制到此处,保持行为等价。
	//  子代理应打开 services/resource_matcher.go 参考,不要重新推导。)

	// 步骤 1: 特殊教室排他检查(理论课 + 排他型教室)
	if requiredType == "" && isSpecialtyRoom(actualType) {
		return MatchResult{
			OK:           false,
			Code:         CodeSpecialtyExclusion,
			Reason:       "theory course cannot use specialty room",
			RequiredType: requiredType,
			ActualType:   actualType,
		}
	}

	// 步骤 2: 教室类型匹配
	if requiredType != "" && requiredType != actualType {
		return MatchResult{
			OK:           false,
			Code:         CodeRoomTypeMismatch,
			Reason:       "room type mismatch",
			RequiredType: requiredType,
			ActualType:   actualType,
		}
	}

	// 步骤 3: 设备匹配
	if missing := missingEquipment(task, room); len(missing) > 0 {
		return MatchResult{
			OK:               false,
			Code:             CodeEquipmentMissing,
			Reason:           "equipment missing",
			RequiredType:     requiredType,
			ActualType:       actualType,
			MissingEquipment: missing,
		}
	}

	return MatchResult{
		OK:           true,
		Code:         MatchOK,
		RequiredType: requiredType,
		ActualType:   actualType,
	}
}

// InferRoomType 推导任务所需教室类型。
// 优先级:task.RequiredRoomType > Course.Category 映射 > 名称推断。
func InferRoomType(task models.TeachingTask, course models.Course) string {
	if task.RequiredRoomType != "" {
		return task.RequiredRoomType
	}
	switch strings.ToUpper(course.Category) {
	case "PE":
		return "GYMNASIUM"
	case "LAB":
		return "LAB"
	case "COMPUTER":
		return "COMPUTER"
	}
	return inferRoomTypeByName(course.Name)
}

// IsSharedVenue 判断教室是否为共享场地(体育馆、图书馆等)。
func IsSharedVenue(room models.Classroom) bool {
	switch strings.ToUpper(room.Type) {
	case "GYMNASIUM", "SHARED":
		return true
	}
	return false
}

// AllowedRooms 从候选列表中筛选所有匹配的教室。
func AllowedRooms(task models.TeachingTask, course models.Course, rooms []models.Classroom) []models.Classroom {
	out := make([]models.Classroom, 0, len(rooms))
	for _, r := range rooms {
		if Match(task, course, r).OK {
			out = append(out, r)
		}
	}
	return out
}

// ExplainRequirement 生成任务的资源需求描述(纯结构化数据)。
func ExplainRequirement(task models.TeachingTask, course models.Course) ResourceRequirement {
	req := ResourceRequirement{
		RoomType:       InferRoomType(task, course),
		Equipment:      parseEquipment(task.RequiredEquipmentJSON),
	}
	if task.RequiredRoomType != "" {
		req.RoomTypeSource = "task"
	} else if course.Category != "" {
		req.RoomTypeSource = "category"
	} else {
		req.RoomTypeSource = "name"
	}
	// IsShared 由 caller 结合具体教室判断,ExplainRequirement 不 populate
	return req
}

// ExplainMismatch 将 MatchResult 转为人类可读的 mismatch 描述。
func ExplainMismatch(result MatchResult) string {
	if result.OK {
		return ""
	}
	switch result.Code {
	case CodeRoomTypeMismatch:
		return "需要 " + result.RequiredType + ",实际 " + result.ActualType
	case CodeEquipmentMissing:
		return "缺少设备:" + strings.Join(result.MissingEquipment, ", ")
	case CodeSpecialtyExclusion:
		return "理论课不能使用专用教室 " + result.ActualType
	}
	return result.Reason
}

// --- 内部辅助 ---

func isSpecialtyRoom(t string) bool {
	switch strings.ToUpper(t) {
	case "LAB", "COMPUTER", "GYMNASIUM":
		return true
	}
	return false
}

func missingEquipment(task models.TeachingTask, room models.Classroom) []string {
	required := parseEquipment(task.RequiredEquipmentJSON)
	if len(required) == 0 {
		return nil
	}
	have := parseEquipment(room.Equipment)
	haveSet := make(map[string]struct{}, len(have))
	for _, e := range have {
		haveSet[strings.ToLower(e)] = struct{}{}
	}
	var missing []string
	for _, r := range required {
		if _, ok := haveSet[strings.ToLower(r)]; !ok {
			missing = append(missing, r)
		}
	}
	return missing
}

func parseEquipment(raw string) []string {
	raw = strings.TrimSpace(raw)
	if raw == "" {
		return nil
	}
	var list []string
	if err := json.Unmarshal([]byte(raw), &list); err != nil {
		// legacy 逗号分隔格式,或坏数据 —— 返回 nil 视为无
		return nil
	}
	return list
}

func inferRoomTypeByName(courseName string) string {
	name := strings.ToLower(courseName)
	switch {
	case strings.Contains(name, "体育") || strings.Contains(name, "sport"):
		return "GYMNASIUM"
	case strings.Contains(name, "实验") || strings.Contains(name, "lab"):
		return "LAB"
	case strings.Contains(name, "计算机") || strings.Contains(name, "程序") || strings.Contains(name, "computer"):
		return "COMPUTER"
	}
	return ""
}
```

```go
package room

import (
	"context"

	"scheduling-system/backend/models"
	"scheduling-system/backend/scheduling/types"
)

// Greedy 是最简单的教室分配算法:按 assignments 顺序,
// 对每个任务尝试所有 classroom,取第一个满足容量 + matcher 的作分配。
// 冲突检测:同 (day, period, classroom) 只能有一个 assignment。
type Greedy struct {
	m Matcher
}

// NewGreedy 构造 Greedy scheduler,注入 Matcher。
func NewGreedy(m Matcher) *Greedy { return &Greedy{m: m} }

// Assign 实现 types.IRoomScheduler。
func (g *Greedy) Assign(ctx context.Context, in types.RoomSchedulingInput, p types.ProgressReporter) (types.RoomSchedulingOutput, error) {
	occupancy := map[slotKey]bool{}
	taskByID := indexTasksByID(in.Tasks)

	var allocations []types.RoomAllocationDraft
	var hints []types.ResourceConflictHint

	for _, a := range in.Assignments {
		task := taskByID[a.TeachingTaskID]

		// 步骤 1: 找容量满足的教室
		var candidates []types.ClassroomView
		for _, c := range in.Classrooms {
			if c.Capacity >= a.TotalStudents {
				candidates = append(candidates, c)
			}
		}
		if len(candidates) == 0 {
			hints = append(hints, hintFor(a, types.ReasonNoCapacity))
			continue
		}

		// 步骤 2: matcher 过滤类型
		var matched []types.ClassroomView
		for _, c := range candidates {
			mr := g.m.Match(taskToModel(task), models.Course{}, roomViewToModel(c))
			if mr.OK {
				matched = append(matched, c)
			}
		}
		if len(matched) == 0 {
			hints = append(hints, hintFor(a, types.ReasonNoMatchingType))
			continue
		}

		// 步骤 3: 找第一个未占用的时间-教室槽(考虑 Span)
		var picked *types.ClassroomView
		for i := range matched {
			c := matched[i]
			if !isOccupied(occupancy, int(a.DayOfWeek), int(a.StartPeriod), a.Span, c.ID) {
				picked = &c
				break
			}
		}
		if picked == nil {
			hints = append(hints, hintFor(a, types.ReasonAllOccupied))
			continue
		}

		markOccupied(occupancy, int(a.DayOfWeek), int(a.StartPeriod), a.Span, picked.ID)
		allocations = append(allocations, types.RoomAllocationDraft{
			LocalRef:    a.LocalRef,
			ClassroomID: picked.ID,
		})
	}

	return types.RoomSchedulingOutput{
		Allocations: allocations,
		Hints:       hints,
	}, nil
}

// slotKey 唯一标识某教室某时刻(单节 period)是否被占用。
type slotKey struct {
	Day, Period int
	RoomID      uint
}

func isOccupied(m map[slotKey]bool, day, startPeriod, span int, roomID uint) bool {
	for p := startPeriod; p < startPeriod+span; p++ {
		if m[slotKey{Day: day, Period: p, RoomID: roomID}] {
			return true
		}
	}
	return false
}

func markOccupied(m map[slotKey]bool, day, startPeriod, span int, roomID uint) {
	for p := startPeriod; p < startPeriod+span; p++ {
		m[slotKey{Day: day, Period: p, RoomID: roomID}] = true
	}
}

func hintFor(a types.TimeAssignmentPlaced, reason types.HintReason) types.ResourceConflictHint {
	return types.ResourceConflictHint{
		TeachingTaskID: a.TeachingTaskID,
		DayOfWeek:      a.DayOfWeek,
		StartPeriod:    a.StartPeriod,
		Span:           a.Span,
		Reason:         reason,
	}
}

func indexTasksByID(tasks []types.TeachingTaskView) map[uint]types.TeachingTaskView {
	m := make(map[uint]types.TeachingTaskView, len(tasks))
	for _, t := range tasks {
		m[t.ID] = t
	}
	return m
}

func taskToModel(t types.TeachingTaskView) models.TeachingTask {
	return models.TeachingTask{
		ID:               t.ID,
		TeacherID:        t.TeacherID,
		CourseID:         t.CourseID,
		RequiredRoomType: t.RequiredRoomType,
	}
}

func roomViewToModel(v types.ClassroomView) models.Classroom {
	return models.Classroom{
		ID:        v.ID,
		Type:      v.Type,
		Floor:     v.Floor,
```

```go
				found = true
				break
			}
		}
		if !found {
			return false
		}
	}
	return true
}

			}
		}
```

```go
// Package orchestrator 组合 time / room / scorer 三个 stage,
// 按 SchedulingMode 装配调度流程。
//
// 见:
//   - 架构 spec §3.5
//   - 执行 spec §6.6
package orchestrator

import (
	"context"
	"errors"
	"time"

	"scheduling-system/backend/scheduling/types"
)

// Orchestrator 是主编排器。
// 内部只持有三个 interface,不 import 任何具体实现 —— 便于生产装配 + fake test。
type Orchestrator struct {
	timeSched  types.ITimeScheduler
	roomSched  types.IRoomScheduler
	scorer     types.IScorer
}

// New 构造 Orchestrator。
func New(time types.ITimeScheduler, room types.IRoomScheduler, scorer types.IScorer) *Orchestrator {
	return &Orchestrator{timeSched: time, roomSched: room, scorer: scorer}
}

// Run 实现 types.ISchedulingOrchestrator。
func (o *Orchestrator) Run(ctx context.Context, req types.OrchestratorRequest, p types.ProgressReporter) (types.OrchestratorResult, error) {
	if !req.Mode.IsValid() {
		return types.OrchestratorResult{}, errors.New("orchestrator: invalid mode " + string(req.Mode))
	}
	start := time.Now()

	// Stage 1: time scheduling
	p.Stage("time", 0)
	timeIn := types.TimeSchedulingInput{
		Tasks:            req.Tasks,
		Teachers:         req.Teachers,
		ClassGroups:      req.ClassGroups,
		LockedSlots:      req.LockedSlots,
		Constraints:      req.Constraints,
		ConstraintWeights: req.ConstraintWeights,
		Seed:             req.Seed,
		Deadline:         req.Deadline,
		SemesterID:       req.SemesterID,
	}
	timeOut, err := o.timeSched.Solve(ctx, timeIn, p)
	if err != nil {
		return types.OrchestratorResult{}, err
	}

	// Stage 2: room scheduling(仅 FULL 模式)
	var allocations []types.RoomAllocationDraft
	if req.Mode.RequiresRoomAssignment() {
		p.Stage("room", 50)
		roomIn := types.RoomSchedulingInput{
			Assignments: draftsToPlaced(timeOut.Assignments, req.Tasks),
			Classrooms:  req.Classrooms,
			Tasks:       req.Tasks,
			Deadline:    req.Deadline,
		}
		roomOut, err := o.roomSched.Assign(ctx, roomIn, p)
		if err != nil {
			return types.OrchestratorResult{}, err
		}
		allocations = roomOut.Allocations
	}

	// Stage 3: score
	p.Stage("score", 90)
	dims := req.Mode.EnabledScoreDimensions()
	scoreBd := o.scorer.Score(timeOut.Assignments, allocations, req.Teachers, req.Classrooms, req.Tasks, dims)

	return types.OrchestratorResult{
		Assignments: timeOut.Assignments,
		Allocations: allocations,
		Score:       scoreBd,
		ElapsedMs:   time.Since(start).Milliseconds(),
	}, nil
}

// draftsToPlaced 把 TimeAssignmentDraft 转成含更多上下文的 TimeAssignmentPlaced,
// 便于 room scheduler 拿到 TotalStudents / RequiredRoomType / AllowedRoomIDs。
func draftsToPlaced(drafts []types.TimeAssignmentDraft, tasks []types.TeachingTaskView) []types.TimeAssignmentPlaced {
	taskByID := make(map[uint]types.TeachingTaskView, len(tasks))
	for _, t := range tasks {
		taskByID[t.ID] = t
	}
	out := make([]types.TimeAssignmentPlaced, 0, len(drafts))
	for i, d := range drafts {
		t := taskByID[d.TeachingTaskID]
		out = append(out, types.TimeAssignmentPlaced{
			LocalRef:         i,
			TeachingTaskID:   d.TeachingTaskID,
			DayOfWeek:        d.DayOfWeek,
			StartPeriod:      d.StartPeriod,
			Span:             d.Span,
			TotalStudents:    t.TotalStudents,
			RequiredRoomType: t.RequiredRoomType,
			AllowedRoomIDs:   t.AllowedRoomIDs,
		})
	}
	return out
}
```
