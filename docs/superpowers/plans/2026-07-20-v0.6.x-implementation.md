```go
// GetEnrichedScheduleEntries 返回展平视图。
// v0.6.0 阶段从旧 schedule_entries 表读取，v0.6.0 模型切换后从 TA+SE JOIN 读取。
func (s *ScheduleQueryService) GetEnrichedScheduleEntries(
	semesterID uint, versionID uint,
) ([]EnrichedScheduleEntry, error) {
	// v0.6.0-pre: 从旧 schedule_entries 表组装
	var entries []models.ScheduleEntry
	q := s.db.Where("semester_id = ?", semesterID).
		Preload("Course").Preload("Teacher").Preload("Classroom").Preload("Classroom.Building").
		Preload("TeachingTask.Classes.ClassGroup")
	if versionID > 0 {
		q = q.Where("schedule_version_id = ?", versionID)
	}
	if err := q.Find(&entries).Error(); err != nil {
		return nil, fmt.Errorf("加载课表条目失败: %w", err)
	}

	result := make([]EnrichedScheduleEntry, 0, len(entries))
	for _, e := range entries {
		enriched := EnrichedScheduleEntry{
			ID:          e.ID,
			DayOfWeek:   int(e.DayOfWeek),
			StartPeriod: int(e.StartPeriod),
			Span:        e.Span,
			Weeks:       e.Weeks,
			TeacherID:   e.TeacherID,
			TeacherName: e.Teacher.Name,
			CourseID:    e.CourseID,
			CourseName:  e.Course.Name,
			CourseCode:  e.Course.Code,
			CourseCredit: e.Course.Credit,
			TeachingTaskID: e.TeachingTaskID,
			SemesterID:  e.SemesterID,
		}
		// 教室字段: IsVirtual 时返回 nil (TIME_ONLY)
		if !e.IsVirtual {
			enriched.ClassroomID = &e.ClassroomID
			name := e.Classroom.Name
			enriched.ClassroomName = &name
			floor := e.Classroom.Floor
			enriched.ClassroomFloor = &floor
			rtype := e.Classroom.RoomType
			enriched.ClassroomType = &rtype
			code := e.Classroom.Code
			enriched.ClassroomCode = &code
			if e.Classroom.Building != nil {
				bname := e.Classroom.Building.Name
				enriched.BuildingName = &bname
			}
		}
		// 班级信息
		if e.TeachingTask != nil {
			for _, cls := range e.TeachingTask.Classes {
				enriched.ClassGroupIDs = append(enriched.ClassGroupIDs, cls.ClassGroupID)
				enriched.ClassGroupNames = append(enriched.ClassGroupNames, cls.ClassGroup.Name)
			}
		}
		result = append(result, enriched)
	}
	return result, nil
}
```

```go
func (a *LegacySASolverAdapter) Solve(
	ctx context.Context,
	input schedtypes.TimeSchedulingInput,
	progress schedtypes.ProgressReporter,
) (schedtypes.TimeSchedulingOutput, error) {
	// 1. 转换 Views → GORM models
	tasks := convertTaskViewsToModels(input.Tasks)
	teachers := convertTeacherViewsToModels(input.Teachers)
	classGroups := convertClassGroupViewsToModels(input.ClassGroups)
	lockedSlots := convertLockedSlots(input.LockedSlots)

	// 2. 虚拟教室 (v0.6.0 临时: 旧 Solver 需要教室参数)
	// v0.6.1 纯 TimeScheduler 删除此 hack
	classrooms := buildVirtualClassroomsForTimeOnly(tasks, classGroups)

	// 3. 配置旧 SA
	saConfig := defaultSAConfig()
	saConfig.TimeOnly = true // 跳过 Match() 调用
	saConfig.IterationsPerTemp = 5000
	saConfig.MaxTimeSeconds = float64(time.Until(input.Deadline).Seconds())
	if saConfig.MaxTimeSeconds <= 0 {
		saConfig.MaxTimeSeconds = 60
	}
	if w, ok := input.ConstraintWeights; ok {
		saConfig.ConstraintWeights = w
	}

	// 4. 委托旧 Solver
	checkDone := func() bool {
		select {
		case <-ctx.Done():
			return true
		default:
			return false
		}
	}
	// 避免阻塞检查，简化: 直接调用
	_ = checkDone

	solver := NewSASolver()
	saResult := solver.SolveMultiRun(
		tasks, teachers, classrooms, classGroups,
		lockedSlots, input.Constraints, input.SemesterID,
		saConfig, 3, nil, nil,
	)

	// 5. 提取时间信息 → TimeAssignmentDraft (丢弃 ClassroomID)
	assignments := make([]schedtypes.TimeAssignmentDraft, 0, len(saResult.Entries))
	for _, e := range saResult.Entries {
		taskID := uint(0)
		if e.TeachingTaskID != nil {
			taskID = *e.TeachingTaskID
		}
		assignments = append(assignments, schedtypes.TimeAssignmentDraft{
			TeachingTaskID: taskID,
			DayOfWeek:      schedtypes.DayOfWeek(e.DayOfWeek),
			StartPeriod:    schedtypes.Period(e.StartPeriod),
			Span:           e.Span,
		})
	}

	return schedtypes.TimeSchedulingOutput{
		Assignments: assignments,
		Diagnostics: nil,
		Iterations:  saResult.Iterations,
		ElapsedMs:   saResult.ElapsedMs,
	}, nil
}
```

```go
func convertTaskViewsToModels(views []schedtypes.TeachingTaskView) []models.TeachingTask {
	out := make([]models.TeachingTask, len(views))
	for i, v := range views {
		out[i] = models.TeachingTask{
			Model:             gorm.Model{ID: v.ID},
			CourseID:          v.CourseID,
			TeacherID:         v.TeacherID,
			SemesterID:        v.SemesterID,
			TotalHours:        v.TotalHours,
			StartWeek:         v.StartWeek,
			EndWeek:           v.EndWeek,
			MaxHoursPerWeek:   v.MaxHoursPerWeek,
			PreferredSpan:     v.PreferredSpan,
			RequiredRoomType:  v.RequiredRoomType,
		}
		out[i].Course = models.Course{
			Model:    gorm.Model{ID: v.CourseID},
			Name:     v.CourseName,
			Category: v.CourseCategory,
			Hours:    v.CourseHours,
		}
		out[i].Teacher = models.Teacher{
			Model:           gorm.Model{ID: v.TeacherID},
			Name:            v.TeacherName,
			PreferNoEarly:   v.PreferNoEarly,
			PreferNoLate:    v.PreferNoLate,
			MaxDaysPerWeek:  v.MaxDaysPerWeek,
			PreferLowFloor:  v.PreferLowFloor,
		}
		classes := make([]models.TeachingTaskClass, len(v.ClassGroupIDs))
		for j, cgID := range v.ClassGroupIDs {
			classes[j] = models.TeachingTaskClass{ClassGroupID: cgID}
		}
		out[i].Classes = classes
	}
	return out
}

	return out
}

		}
	}

			DayOfWeek:   models.DayOfWeek(l.DayOfWeek),
			StartPeriod: models.Period(l.StartPeriod),
			Span:        l.Span,
		}
	}
	return out
}
```

```go
// backend/services/adapter_legacy_room.go
package services

import (
	"context"
	"sort"

	schedtypes "scheduling-system/backend/scheduling/types"
)

// LegacyRoomAllocatorAdapter 实现 types.IRoomScheduler，
// 从旧 ScheduleEntry 提取教室分配信息。
// v0.6.1 被 scheduling/room/ 的 Greedy RoomScheduler 替换。

type LegacyRoomAllocatorAdapter struct{}

func NewLegacyRoomAllocatorAdapter() *LegacyRoomAllocatorAdapter {
	return &LegacyRoomAllocatorAdapter{}
}

var _ schedtypes.IRoomScheduler = (*LegacyRoomAllocatorAdapter)(nil)

func (a *LegacyRoomAllocatorAdapter) Assign(
	ctx context.Context,
	input schedtypes.RoomSchedulingInput,
	progress schedtypes.ProgressReporter,
) (schedtypes.RoomSchedulingOutput, error) {
	// v0.6.0 阶段: 简单贪心分配。
	// 按 AllowedRoomIDs 数量升序 (最受限的优先)。
	sorted := make([]schedtypes.TimeAssignmentPlaced, len(input.Assignments))
	copy(sorted, input.Assignments)
	sort.Slice(sorted, func(i, j int) bool {
		return len(sorted[i].AllowedRoomIDs) < len(sorted[j].AllowedRoomIDs)
	})

	// 教室占用追踪
	type slotKey struct{ roomID uint; day int; period int }
	occupied := make(map[slotKey]bool)
	roomByID := make(map[uint]schedtypes.ClassroomView, len(input.Classrooms))
	for _, c := range input.Classrooms {
		roomByID[c.ID] = c
	}
	taskByID := make(map[uint]schedtypes.TeachingTaskView, len(input.Tasks))
	for _, t := range input.Tasks {
		taskByID[t.ID] = t
	}

	allocations := make([]schedtypes.RoomAllocationDraft, 0, len(sorted))
	hints := make([]schedtypes.ResourceConflictHint, 0)

	for _, placed := range sorted {
		// 尝试 AllowedRoomIDs 中的每个教室
		assigned := false
		for _, roomID := range placed.AllowedRoomIDs {
			room, ok := roomByID[roomID]
			if !ok {
				continue
			}
			// 检查容量
			if room.Capacity < placed.TotalStudents && room.Type != "GYM" {
				continue
			}
			// 检查时间占用
			conflict := false
			for p := placed.StartPeriod; p < placed.StartPeriod+schedtypes.Period(placed.Span); p++ {
				key := slotKey{roomID, int(placed.DayOfWeek), int(p)}
				if occupied[key] {
					conflict = true
					break
				}
			}
			if conflict {
				continue
			}
			// 分配
			for p := placed.StartPeriod; p < placed.StartPeriod+schedtypes.Period(placed.Span); p++ {
				occupied[slotKey{roomID, int(placed.DayOfWeek), int(p)}] = true
			}
			allocations = append(allocations, schedtypes.RoomAllocationDraft{
				LocalRef:    placed.LocalRef,
				ClassroomID: roomID,
			})
			assigned = true
			break
		}
		if !assigned {
			hints = append(hints, schedtypes.ResourceConflictHint{
				TeachingTaskID: placed.TeachingTaskID,
				DayOfWeek:      placed.DayOfWeek,
				StartPeriod:    placed.StartPeriod,
				Span:           placed.Span,
				Reason:         schedtypes.ReasonAllOccupied,
				Detail:         "所有可用教室在该时段已被占用",
			})
		}
	}

	return schedtypes.RoomSchedulingOutput{
		Allocations: allocations,
		Hints:       hints,
	}, nil
}
```

```go
// backend/services/adapter_legacy_score.go
package services

import (
	schedtypes "scheduling-system/backend/scheduling/types"
)

// LegacyScorerAdapter 实现 types.IScorer，
// 内部委托给旧 ScoringService.ScoreSchedule。
// v0.6.1 被 scheduling/score/ 的 4-bucket Scorer 替换。

type LegacyScorerAdapter struct {
	scoringService *ScoringService
}

func NewLegacyScorerAdapter() *LegacyScorerAdapter {
	return &LegacyScorerAdapter{scoringService: NewScoringService()}
}

var _ schedtypes.IScorer = (*LegacyScorerAdapter)(nil)

func (a *LegacyScorerAdapter) Score(
	assignments []schedtypes.TimeAssignmentDraft,
	allocations []schedtypes.RoomAllocationDraft,
	teachers []schedtypes.TeacherView,
	classrooms []schedtypes.ClassroomView,
	tasks []schedtypes.TeachingTaskView,
	dims []string,
) schedtypes.ScoreBreakdown {
	// 转换 Draft → 旧 ScheduleEntry (评分接口需要)
	// v0.6.0 阶段仅限 Orchestrator 内部使用
	// 评分委托给旧 ScoringService，但通过 v0.6.0 数据模型返回 ScoreBreakdown

	// 构建评分上下文
	// (简化实现: 使用默认约束 + Mode 推导的 dims)
	teacherModels := convertTeacherViewsToModels(teachers)
	classroomModels := convertClassroomViewsToModels(classrooms)
	taskModels := convertTaskViewsToModels(tasks)

	// 重建旧格式 entries
	entries := make([]ScheduleEntry, 0, len(assignments))
	allocByRef := make(map[int]schedtypes.RoomAllocationDraft, len(allocations))
	for _, a := range allocations {
		allocByRef[a.LocalRef] = a
	}
	taskByID := make(map[uint]schedtypes.TeachingTaskView, len(tasks))
	for _, t := range tasks {
		taskByID[t.ID] = t
	}

	for i, draft := range assignments {
		taskView := taskByID[draft.TeachingTaskID]
		entry := ScheduleEntry{
			CourseID:    taskView.CourseID,
			TeacherID:   taskView.TeacherID,
			DayOfWeek:   models.DayOfWeek(draft.DayOfWeek),
			StartPeriod: models.Period(draft.StartPeriod),
			Span:        draft.Span,
			Weeks:       fmt.Sprintf("%d-%d", taskView.StartWeek, taskView.EndWeek),
		}
		entry.TeachingTaskID = &draft.TeachingTaskID
		if alloc, ok := allocByRef[i]; ok {
			entry.ClassroomID = alloc.ClassroomID
		}
		entries = append(entries, entry)
	}

	scoringCtx := NewScoringContext(dims, nil, taskModels) // 简化
	breakdown := a.scoringService.ScoreSchedule(entries, teacherModels, classroomModels, scoringCtx)

	// 转换为新版 ScoreBreakdown
	return schedtypes.ScoreBreakdown{
		Total:      breakdown.Total,
		FinalTotal: breakdown.FinalTotal,
		Completeness:  breakdown.Completeness,
		PlacedSessions:  breakdown.PlacedSessions,
		ExpectedSessions: breakdown.ExpectedSessions,
		EnabledDimensions: dims,
	}
}
```

```go
	}
```

```go
	q := s.db.Where("semester_id = ?", semesterID)

	}
	if err := q.Find(&tas).Error(); err != nil {

	}
	
	// 批量查 ScheduleEntry (FULL 模式)
	taIDs := make([]uint, len(tas))
	for i, ta := range tas {

	}
	var ses []models.ScheduleEntry
	if len(taIDs) > 0 {

	for _, se := range ses {

	ttIDs := make([]uint, len(tas))
	for i, ta := range tas {

	}

	for _, tt := range tts {
		ttByID[tt.ID] = tt
```

```go
func (ts *TimeScheduler) Solve(
	ctx context.Context,
	input schedtypes.TimeSchedulingInput,
	progress schedtypes.ProgressReporter,
) (schedtypes.TimeSchedulingOutput, error) {
	start := time.Now()
	rng := rand.New(rand.NewSource(input.Seed))

	// 构建时间调度上下文 (不含教室/roomOcc)
	sctx := newTimeContext(input, rng)

	// 构建初始解 (仅分配时间)
	sctx.buildInitialTime()

	// 初始评分 (3 buckets: time + teacher + student)
	bestScore := sctx.computeScore()
	bestAssignments := sctx.cloneAssignments()

	// SA 主循环
	temp := ts.config.InitialTemp
	iter := 0
	deadline := start.Add(time.Duration(ts.config.MaxTimeSeconds) * time.Second)

	for temp > ts.config.MinTemp && time.Now().Before(deadline) {
		for i := 0; i < ts.config.IterationsPerTemp; i++ {
			select {
			case <-ctx.Done():
				goto done
			default:
			}
			iter++

			newScore := sctx.tryTimeNeighbor(bestScore, temp, rng)
			if newScore > bestScore {
				bestScore = newScore
				bestAssignments = sctx.cloneAssignments()
			}
		}
		temp *= ts.config.CoolingRate

		if iter%100 == 0 {
			progress.Iteration(iter, 0, bestScore, bestScore, temp)
		}
	}
done:

	return schedtypes.TimeSchedulingOutput{
		Assignments: bestAssignments,
		Iterations:  iter,
		ElapsedMs:   time.Since(start).Milliseconds(),
	}, nil
}
```

```go
func (sctx *timeContext) tryTimeNeighbor(currentScore float64, temp float64, rng *rand.Rand) float64 {
	if rng.Float64() < 0.7 {
		return sctx.tryMoveTime(currentScore, temp, rng)
	}
	return sctx.trySwapTime(currentScore, temp, rng)
}

func (sctx *timeContext) tryMoveTime(currentScore float64, temp float64, rng *rand.Rand) float64 {
	// 随机选一个 entry
	idx := rng.Intn(len(sctx.assignments))
	entry := sctx.assignments[idx]

	// 保存旧状态
	oldDay, oldStart := entry.DayOfWeek, entry.StartPeriod

	// 移除占用
	sctx.removeTimeOccupancy(entry)

	// 随机选新时间 (仅检查: locked slots, teacher busy, class busy, teacher unavailable)
	// 不检查 roomOcc, 不遍历教室
	newDay := schedtypes.DayOfWeek(rng.Intn(7))
	validStarts := schedtypes.ValidStartsForSpan(entry.Span)
	newStart := validStarts[rng.Intn(len(validStarts))]

	entry.DayOfWeek = newDay
	entry.StartPeriod = newStart

	if sctx.hasTimeConflict(entry) {
		// 恢复
		entry.DayOfWeek = oldDay
		entry.StartPeriod = oldStart
		sctx.addTimeOccupancy(entry)
		return currentScore
	}

	sctx.addTimeOccupancy(entry)
	newScore := sctx.computeScore()
	delta := newScore - currentScore

	if delta > 0 || rng.Float64() < math.Exp(delta/temp) {
		return newScore
	}

	// 拒绝，恢复
	sctx.removeTimeOccupancy(entry)
	entry.DayOfWeek = oldDay
	entry.StartPeriod = oldStart
	sctx.addTimeOccupancy(entry)
	return currentScore
}
```

```go
package score

import (

type Scorer struct{}

func NewScorer() *Scorer { return &Scorer{} }

var _ schedtypes.IScorer = (*Scorer)(nil)

func (s *Scorer) Score(
	assignments []schedtypes.TimeAssignmentDraft,
	allocations []schedtypes.RoomAllocationDraft,
	teachers []schedtypes.TeacherView,
	classrooms []schedtypes.ClassroomView,
	tasks []schedtypes.TeachingTaskView,
	dims []string,
) schedtypes.ScoreBreakdown {
	result := schedtypes.ScoreBreakdown{
		EnabledDimensions: dims,
		PlacedSessions:    len(assignments),
	}

	// 计算 expectedSessions
	expected := 0
	for _, t := range tasks {
		plan := schedtypes.ResolveSessionPlan(t.TotalHours, t.StartWeek, t.EndWeek, t.MaxHoursPerWeek, t.PreferredSpan)
		expected += plan.SessionsPerWeek()
	}
	result.ExpectedSessions = expected

	perBucketMax := 100.0 / float64(len(dims))
	result.PerBucketMax = perBucketMax

	// 按 dims 顺序计算各 bucket
	for _, dim := range dims {
		switch dim {
		case "time":
			bucket := s.scoreTimeBucket(assignments, tasks, teachers)
			bucket.Max = perBucketMax
			result.Time = &bucket
		case "teacher":
			bucket := s.scoreTeacherBucket(assignments, tasks, teachers)
			bucket.Max = perBucketMax
			result.Teacher = &bucket
		case "student":
			bucket := s.scoreStudentBucket(assignments, tasks)
			bucket.Max = perBucketMax
			result.Student = &bucket
		case "resource":
			bucket := s.scoreResourceBucket(allocations, classrooms, teachers)
			bucket.Max = perBucketMax
			result.Resource = &bucket
		}
	}

	// 计算 Total
	var total float64
	for _, b := range []*schedtypes.ScoreBucket{result.Time, result.Teacher, result.Student, result.Resource} {
		if b != nil {
			total += b.Value
		}
	}
	result.Total = total

	// Completeness
	if result.ExpectedSessions > 0 {
		ratio := float64(result.PlacedSessions) / float64(result.ExpectedSessions)
		result.Completeness = ratio * (0.5 + 0.5*ratio)
	}
	result.FinalTotal = result.Total * result.Completeness

	return result
}
```
